   37 insert1 :: Integer -> BinaryTree -> BinaryTree
   38 insert1 i Leaf = Leaf
   39 insert1 i (Branch v l r) 
   40   | i < v = Branch v (insert1 i l) r
   41   | otherwise =  Branch v l (insert1 i r)
   42 
   43 insert2 :: Integer -> BinaryTree -> BinaryTree
   44 insert2 i Leaf = Branch i Leaf Leaf
   45 insert2 i (Branch v l r) 
   46   | i < v = Branch v (insert2 i l) r
   47   | i > v = Branch v l (insert2 i r)

   49 insert3 :: Integer -> BinaryTree -> BinaryTree
